{() => {
const runningOS = navigator.userAgent;
  
const getOperatingSystem = (() => {
      let oSystem=navigator.userAgent;
      let osString="";
      if (typeof window === "undefined") {
       //If the code is not running in the browser (e.g., server-side), return null - we don't want errors
      return null;
    }
        if (oSystem.search("Win") !== -1) {
      oSystem = "Windows";
    } else if (oSystem.search("Mac") !== -1) {
      osString = "Mac";
    } else {
      return "Unknown";
    }
return osString;
  })();

  
  <p>In order to run Remote Incident Manager, your computer should meet or exceed the following minimum requirements</p>
  if (getOperatingSystem === "Windows") {
    return (
      <>
        <p>Windows Requirements:</p>
        <ul>
          <li>Windows 7 SP1 or later</li>
          <li>Windows Server 2012 or later</li>
          <li>At least 4GB of RAM</li>
          <li>Admin privileges for initial installation (not required for automatic updates)</li>
        </ul>
      </>
    );
  }

  else if (getOperatingSystem=== "Mac") {
    return (
      <>
        <p>Mac Requirements:</p>
        <ul>
          <li>Intel or Apple Silicon</li>
          <li>Mac OS Ventura</li>
          <li>At least 4GB of RAM</li>
          <li>Admin privileges for initial installation (not required for automatic updates)</li>
        </ul>
      </>
    );
  }

  return ("That should be the system requirements for " + getOperatingSystem + " as far as we can tell"); // Handle the case for "Unknown" or other platforms if needed
}}
