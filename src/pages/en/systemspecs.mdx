title: "System Requirements"
---

import React from 'react';

const SystemRequirementsPage = () => {
  const oSystem = navigator.userAgent;

  const getOperatingSystem = () => {
    if (oSystem.search(/Win/i) !== -1) {
      return 'Windows';
    } else if (oSystem.search(/Mac/i) !== -1) {
      return 'Mac';
    } else {
      return 'Unknown';
    }
  };

  const operatingSystem = getOperatingSystem();

  return (
    <>
      <h1>{operatingSystem} System Requirements</h1>

      {operatingSystem === 'Windows' && (
        <>
          <h2>Windows Requirements:</h2>
          <ul>
            <li>Windows 7 SP1 or later</li>
            <li>Windows Server 2012 or later</li>
            <li>At least 4GB of RAM</li>
            <li>Admin privileges for initial installation (not required for automatic updates)</li>
          </ul>
        </>
      )}

      {operatingSystem === 'Mac' && (
        <>
          <h2>Mac Requirements:</h2>
          <ul>
            <li>Intel or Apple Silicon</li>
            <li>Mac OS Ventura</li>
            <li>At least 4GB of RAM</li>
            <li>Admin privileges for initial installation (not required for automatic updates)</li>
          </ul>
        </>
      )}

      {operatingSystem === 'Unknown' && (
        <p>We were not able to produce system requirments, you may be running an operating system RIM does not suppor, or there is an error with the rendering of the page. Sorry!</p>
      )}
    </>
  );
};

export default SystemRequirementsPage;
